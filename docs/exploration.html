vec <- names(data[ , grepl( "Have you ever plotted" , names(data))])
do_vec <- function(vec, data){
  t1 <- table(data[vec[1]])
  for (i in 2:length(vec)) {
    t <- table(data[vec[i]])
    t1 <- rbind(t1, t)
  }
  rownames(t1) <- array(paste0("t_", 1:length(vec)))
  return(t1)
}

do_vec <- function(vec, data, plot_names){
